QEMU vhost-vdpa Shadow Virtqueue 详解

发表于2026-03-09|更新于2026-03-18|内核与虚拟化
|浏览量:

Shadow Virtqueue（SVQ）是 QEMU vhost-vdpa 中的一个关键机制，用于在 Guest 和 vDPA 后端硬件之间插入一个"影子"层，实现地址翻译、流量拦截与回调处理。本文梳理 SVQ 的整体架构、文件描述符体系以及各方向的通知链路。

一、整体架构

SVQ 将原本 Guest 直通 Backend 的 virtqueue 路径拦截，在中间插入 QEMU 作为代理：

Guest
  |  (svq_kick: Guest 写入，QEMU 监听)
  v
QEMU / SVQ  <----(hdev_call: Backend 完成后通知 QEMU)
  |  (hdev_kick: QEMU 向 Backend 发起通知)
  v
Backend 硬件
  |
  `--------> (svq_call: QEMU 注入中断，通知 Guest)
             Guest

SVQ 维护四个文件描述符，分别对应两侧的 kick/call：

FD 方向 说明
svq->svq_kick Guest → QEMU 接管原有 Guest kick fd，QEMU 监听
svq->hdev_kick QEMU → Backend QEMU 新建，用于通知 Backend
svq->hdev_call Backend → QEMU QEMU 新建，Backend 完成时触发
svq->svq_call QEMU → Guest 保留原有 call fd，QEMU 发起中断

二、SVQ 的创建与启动

2.1 创建（vhost_svq_new）

创建阶段在设备初始化时完成，绑定回调 ops：

vhost_vdpa_init                          [vhost-vdpa.c]
└── vhost_vdpa_init_svq(dev, v)
    └── vhost_svq_new(ops, ops_opaque)   [vhost-shadow-virtqueue.c]
        # 分配 SVQ 实例
        # 初始化 svq_kick 为 VHOST_FILE_UNBIND（占位）
        # 绑定 ops / ops_opaque
VhostShadowVirtqueue *vhost_svq_new(const VhostShadowVirtqueueOps *ops,
                                    void *ops_opaque)
{
    VhostShadowVirtqueue *svq = g_new0(VhostShadowVirtqueue, 1);

    event_notifier_init_fd(&svq->svq_kick, VHOST_FILE_UNBIND);
    svq->ops = ops;
    svq->ops_opaque = ops_opaque;
    return svq;
}

ops 决定"做什么"，ops_opaque 是回调函数的上下文指针。

2.2 启动（vhost_svq_start）

设备启动阶段完成 FD 初始化与事件处理注册：

vhost_vdpa_dev_start                             [vhost-vdpa.c]
└── vhost_vdpa_svqs_start(dev)
    ├── vhost_vdpa_svq_setup(dev, svq, i, &err)
    │   └── vhost_vdpa_svq_set_fds(dev, svq, idx, errp)
    │       ├── event_notifier_init(&svq->hdev_kick, 0)   # 新建 hdev_kick
    │       ├── event_notifier_init(&svq->hdev_call, 0)   # 新建 hdev_call
    │       ├── vhost_vdpa_set_vring_dev_kick(dev, &file) # 告知 Backend kick fd
    │       └── vhost_vdpa_set_vring_dev_call(dev, &file) # 告知 Backend call fd
    └── vhost_svq_start(svq, dev->vdev, vq, iova_tree)   [vhost-shadow-virtqueue.c]
        └── event_notifier_set_handler(&svq->hdev_call, vhost_svq_handle_call)
            # 注册 hdev_call 的事件回调

三、四条通知链路详解

3.1 svq_kick：Guest → QEMU

原本 Guest 直接 kick Backend，SVQ 接管该 FD，QEMU 转为监听方。

设置链路：

vhost_vdpa_set_vring_kick                           [vhost-vdpa.c]
└── vhost_svq_set_svq_kick_fd(svq, file->fd)        [vhost-shadow-virtqueue.c]
    ├── event_notifier_init_fd(svq_kick, svq_kick_fd)  # 绑定 Guest 的 kick fd
    ├── event_notifier_set(svq_kick)                   # 触发一次，处理潜在的漏通知
    └── event_notifier_set_handler(svq_kick,           # 注册监听回调
            vhost_handle_guest_kick_notifier)
void vhost_svq_set_svq_kick_fd(VhostShadowVirtqueue *svq, int svq_kick_fd)
{
    EventNotifier *svq_kick = &svq->svq_kick;
    bool poll_stop = VHOST_FILE_UNBIND != event_notifier_get_fd(svq_kick);
    bool poll_start = svq_kick_fd != VHOST_FILE_UNBIND;

    if (poll_stop) {
        event_notifier_set_handler(svq_kick, NULL);  /* 先摘除旧的监听 */
    }

    event_notifier_init_fd(svq_kick, svq_kick_fd);

    if (poll_start) {
        event_notifier_set(svq_kick);
        event_notifier_set_handler(svq_kick, vhost_handle_guest_kick_notifier);
    }
}

触发链路：

[Guest 写入 kick fd]
    |
    v
vhost_handle_guest_kick_notifier(svq_kick)
└── vhost_handle_guest_kick(svq)
    └── do {
            elem = virtqueue_pop(svq->vq, ...)   # 从 vq 弹出请求
            if (svq->ops)
                svq->ops->avail_handler(...)     # 调用自定义回调（如地址翻译）
            else
                vhost_svq_add_element(svq, elem) # 默认：直接送入影子队列
        } while (!virtio_queue_empty(svq->vq))

3.2 hdev_kick：QEMU → Backend

QEMU 处理完请求后，通过新建的 hdev_kick fd 通知 Backend 处理。

设置链路（见 2.2 启动流程，vhost_vdpa_svq_set_fds 中完成）

触发链路：

vhost_svq_add_element(svq, elem)
└── vhost_svq_add(svq, elem, ...)
    └── vhost_svq_kick(svq)
        ├── smp_mb()                              # 内存屏障，保证 avail 可见
        ├── [检查 VIRTIO_RING_F_EVENT_IDX]
        │   ├── (有) vring_need_event(avail_event, ...) → needs_kick
        │   └── (无) !(used->flags & VRING_USED_F_NO_NOTIFY) → needs_kick
        └── if (needs_kick)
                event_notifier_set(&svq->hdev_kick)  # 通知 Backend
static void vhost_svq_kick(VhostShadowVirtqueue *svq)
{
    bool needs_kick;

    smp_mb();

    if (virtio_vdev_has_feature(svq->vdev, VIRTIO_RING_F_EVENT_IDX)) {
        uint16_t avail_event = le16_to_cpu(
                *(uint16_t *)(&svq->vring.used->ring[svq->vring.num]));
        needs_kick = vring_need_event(avail_event,
                                      svq->shadow_avail_idx,
                                      svq->shadow_avail_idx - 1);
    } else {
        needs_kick = !(svq->vring.used->flags & cpu_to_le16(VRING_USED_F_NO_NOTIFY));
    }

    if (!needs_kick) {
        return;
    }

    event_notifier_set(&svq->hdev_kick);
}

3.3 hdev_call：Backend → QEMU

Backend 处理完请求后，写入 hdev_call fd，QEMU 的回调被触发。

设置链路（见 2.2，vhost_svq_start 中注册回调）

触发链路：

[Backend 写入 hdev_call fd]
    |
    v
vhost_svq_handle_call(&svq->hdev_call)
└── vhost_svq_flush(svq, true)
    └── do {
            while (true) {
                elem = vhost_svq_get_buf(svq, &len)  # 从影子队列取出已完成请求
                if (!elem) break
                virtqueue_fill(vq, elem, len, i++)   # 填回 Guest virtqueue
            }
            virtqueue_flush(vq, i)                   # 刷新，让 Guest 可见
            event_notifier_set(&svq->svq_call)       # 触发 svq_call，通知 Guest
        } while (!vhost_svq_enable_notification(svq))

3.4 svq_call：QEMU → Guest

QEMU 在完成 vhost_svq_flush 后，通过保留的原始 svq_call fd 向 Guest 注入中断。

设置链路：

vhost_vdpa_set_vring_call                          [vhost-vdpa.c]
└── vhost_svq_set_svq_call_fd(svq, file->fd)       [vhost-shadow-virtqueue.c]
    └── event_notifier_init_fd(&svq->svq_call, call_fd)
        # 绑定原始 Guest call fd

触发链路：

vhost_svq_flush(svq, ...)
└── event_notifier_set(&svq->svq_call)   # 注入 Guest 中断
    |
    v
 [Guest 收到 virtio 完成中断]
static void vhost_svq_flush(VhostShadowVirtqueue *svq,
                            bool check_for_avail_queue)
{
    do {
        while (true) {
            g_autofree VirtQueueElement *elem = vhost_svq_get_buf(svq, &len);
            if (!elem) {
                break;
            }
            virtqueue_fill(vq, elem, len, i++);  /* 填回 Guest virtqueue */
        }

        virtqueue_flush(vq, i);                  /* 刷新，确保 Guest 可见 */
        event_notifier_set(&svq->svq_call);      /* 注入 Guest 中断 */

    } while (!vhost_svq_enable_notification(svq));
}

3.5 完整数据流总结

Guest
  |
  | [写 svq_kick fd]
  v
vhost_handle_guest_kick_notifier
  └── virtqueue_pop → ops->avail_handler / vhost_svq_add_element
                                |
                                | [vhost_svq_kick]
                                v
                          event_notifier_set(hdev_kick)
                                |
                                | [Backend 处理]
                                v
                          [Backend 写 hdev_call fd]
                                |
                                v
                    vhost_svq_handle_call
                      └── vhost_svq_flush
                            ├── vhost_svq_get_buf
                            ├── virtqueue_fill / virtqueue_flush
                            └── event_notifier_set(svq_call)
                                        |
                                        | [注入中断]
                                        v
                                      Guest
